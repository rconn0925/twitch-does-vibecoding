export function sanitizeRepoName(title: string, now: () => Date): string {
  let slug = title
    .toLowerCase()
    .replaceAll(/[^a-z0-9]+/g, "-")
    .replaceAll(/^-+|-+$/g, "");
  if (slug.length > 80) {
    slug = slug.slice(0, 80).replaceAll(/-+$/g, "");
  }
  return slug.length > 0 ? slug : datedFallback(now());
}
